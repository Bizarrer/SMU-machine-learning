According to Pareto principle, a 80/20 split of the dataset into training(80%) and testing(20%) is very popular in machine learning, there is a reason for that: depending on the total number of observations, absolute number of training data(80%) can be very small, for example, if you have only 100 or less instances, there will be 80 or less training data, the parameters obtained from training data will have big variation, the model will not accurately predict the testing results. However, if the dataset size is very big,for example, 10,000, then a 80:20 or 90:10 does not matter, since either split will give you small variance for the model, the only thing you need to consider at this time is to use less data to avoid excessive computation.

2.1 Steepest descent, Stochastic gradient descent, and Newton's method¶

When we use steepest descent, stochastic gradient descent, we update the gradient calculation by three senarios:

1) L1 regularization

2) L2 regularization

3) Both L1 regularization and L2 regularization

There are two ways for Newton's method, Hessian and BFGS.Because Hession has some problems(cited from the instructor's ML notebooks):

1) Highly dependent on starting point

2) Jumps can get really random

3) Near saddle points, inverse hessian unstable

4) Hessian not always invertible or invertible with correct numerical precision

We chose BFGS here to finish Newton's method. we set the initial values of C1 and C2 equal to 0.001.

# from last time, our logistic regression algorithm is given by (including everything we previously had):
class L1Regularization:
    def __init__(self, eta, iterations=100, C1=0.001, method='steepest'):
        self.eta = eta
        self.iters = iterations
        self.C = C1
        self.method = method
        # internally we will store the weights as self.w_ to keep with sklearn conventions
        
    def __str__(self):
        if(hasattr(self,'w_')):
            return 'Binary Logistic Regression Object with coefficients:\n'+ str(self.w_) # is we have trained the object
        else:
            return 'Untrained Binary Logistic Regression Object'
        
    # convenience, private:
    @staticmethod
    def _add_bias(X):
        return np.hstack((np.ones((X.shape[0],1)),X)) # add bias term
    
    @staticmethod
    def _sigmoid(theta):
        # increase stability, redefine sigmoid operation
        return expit(theta) #1/(1+np.exp(-theta))
 
    @staticmethod
    def _sgn(w):
        if w>0:
            return 1
        elif w<0:
            return -1
        else:
            return 0
        
    @staticmethod
    def objective_function(w,X,y,C1):
        g = expit(X @ w)
        return -np.sum(np.log(g[y==1]))-np.sum(np.log(1-g[y==0])) + C1*sum(w**2) #-np.sum(y*np.log(g)+(1-y)*np.log(1-g))

    @staticmethod
    def objective_gradient(w,X,y,C1):
        g = expit(X @ w)
        ydiff = y-g # get y difference
        gradient = np.mean(X * ydiff[:,np.newaxis], axis=0)
        gradient = gradient.reshape(w.shape)
        gradient[1:] += -2 * w[1:] * C1
        return -gradient
    
        #steepest descent
    def _steepest_descent_get_gradient(self,X,y):
        ydiff = y-self.predict_proba(X,add_bias=False).ravel() # get y difference
        gradient = np.mean(X * ydiff[:,np.newaxis], axis=0) # make ydiff a column vector and multiply through
        gradient = gradient.reshape(self.w_.shape)
        
        # using both L1 and L2 regularization
        sgn_v = np.vectorize(self._sgn)
        # new gradient update method
        gradient[1:] += -2 * self.w_[1:] * self.C 
        return gradient

        # stochastic gradient calculation 
    def _stochastic_gradient_get_gradient(self,X,y):
        idx = int(np.random.rand()*len(y)) # grab random instance
        ydiff = y[idx]-self.predict_proba(X[idx],add_bias=False) # get y difference (now scalar)
        gradient = X[idx] * ydiff[:,np.newaxis] # make ydiff a column vector and multiply through
        gradient = gradient.reshape(self.w_.shape)
  
        # using L1 regularization
        sgn_v = np.vectorize(self._sgn)
        # new gradient update method
        gradient[1:] += -2 * self.w_[1:] * self.C 
        return gradient
    
    # public:
    def predict_proba(self,X,add_bias=True):
        # add bias term if requested
        Xb = self._add_bias(X) if add_bias else X
        return self._sigmoid(Xb @ self.w_) # return the probability y=1
    
    def predict(self,X):
        return (self.predict_proba(X)>0.5) #return the actual prediction
    
    def fit(self, X, y):
        Xb = self._add_bias(X) # add bias term
        num_samples, num_features = Xb.shape
        
        self.w_ = np.zeros((num_features,1)) # init weight vector to zeros
        
        # for as many as the max iterations
        # choose differt gradient caculation function
        for _ in range(self.iters):
            if self.method =='steepest':
                gradient = self._steepest_descent_get_gradient(Xb,y)
            
            if self.method =='stochastic':
                gradient = self._stochastic_gradient_get_gradient(Xb,y)

            if self.method =='newton':
                self.w_ = fmin_bfgs(self.objective_function, # what to optimize
                            np.zeros((num_features,1)), # starting point
                            fprime=self.objective_gradient, # gradient function
                            args=(Xb,y,self.C), # extra args for gradient and objective function
                            gtol=1e-03, # stopping criteria for gradient, |v_k|
                            maxiter=self.iters, # stopping criteria iterations
                            disp=False)
        
                self.w_ = self.w_.reshape((num_features,1))  
                continue
                 
            self.w_ += gradient*self.eta # multiply by learning rate

        ydiff = y-self.predict_proba(X,add_bias=False).ravel() # get y difference
        gradient = np.mean(X * ydiff[:,np.newaxis], axis=0) # make ydiff a column vector and multiply through
        gradient = gradient.reshape(self.w_.shape)
        
        # using both L1 and L2 regularization
        sgn_v = np.vectorize(self._sgn)
        # new gradient update method
        gradient[1:] += -2 * self.w_[1:] * self.C1 - self.C2 * sgn_v(self.w_[1:])
        return gradient

        # stochastic gradient calculation 
    def _stochastic_gradient_get_gradient(self,X,y):
        idx = int(np.random.rand()*len(y)) # grab random instance
        ydiff = y[idx]-self.predict_proba(X[idx],add_bias=False) # get y difference (now scalar)
        gradient = X[idx] * ydiff[:,np.newaxis] # make ydiff a column vector and multiply through
        gradient = gradient.reshape(self.w_.shape)
  
        # using both L1 and L2 regularization
        sgn_v = np.vectorize(self._sgn)
        # new gradient update method
        gradient[1:] += -2 * self.w_[1:] * self.C1 - self.C2 * sgn_v(self.w_[1:])
        return gradient
    
    # public:
    def predict_proba(self,X,add_bias=True):
        # add bias term if requested
        Xb = self._add_bias(X) if add_bias else X
        return self._sigmoid(Xb @ self.w_) # return the probability y=1
    
    def predict(self,X):
        return (self.predict_proba(X)>0.5) #return the actual prediction
    
    def fit(self, X, y):
        Xb = self._add_bias(X) # add bias term
        num_samples, num_features = Xb.shape
        
        self.w_ = np.zeros((num_features,1)) # init weight vector to zeros
        
        # for as many as the max iterations
        # choose differt gradient caculation function
        for _ in range(self.iters):
            if self.method =='steepest':
                gradient = self._steepest_descent_get_gradient(Xb,y)
            
            if self.method =='stochastic':
                gradient = self._stochastic_gradient_get_gradient(Xb,y)

            if self.method =='newton':
                self.w_ = fmin_bfgs(self.objective_function, # what to optimize
                            np.zeros((num_features,1)), # starting point
                            fprime=self.objective_gradient, # gradient function
                            args=(Xb,y,self.C1), # extra args for gradient and objective function
                            gtol=1e-03, # stopping criteria for gradient, |v_k|
                            maxiter=self.iters, # stopping criteria iterations
                            disp=False)
        
                self.w_ = self.w_.reshape((num_features,1))  
                continue
                 
            self.w_ += gradient*self.eta # multiply by learning rate

        self.iters = iterations
        self.C2 = C2

3 Deployment¶

We would advise to use scikit-learn logistic regression to be used in a deployed machine learning model, because the implementation time of scikit-learn is way way shorter than our model, though the accuracy of scikit-learn is only a little bit higher than ours.

The most important reason that we recommend you to use scikit-learn is that if you use our own model,it's a little bit of arduous for you to find the best optimazation method and the values of best regularization and best cost term. But if you use scikit-learn, it will have already given you the default wanted parameters, finally, implementing logistic regression model will be much faster and more accurate.

In conclusion, we recommend using scikit-learn implementation for deployment in a machine learning model.

Here we have a question why scikit-learn runs so fast?
Accodeing to https://scikit-learn.org/stable/developers/performance.html (citation)
In general, the scikit-learn project emphasizes the readability of the source code to make it easy for the project users to dive into the source code so as to understand how the algorithm behaves on their data but also for ease of maintainability (by the developers). When implementing a new algorithm is thus recommended to start implementing it in Python using Numpy and Scipy by taking care of avoiding looping code using the vectorized idioms of those libraries. In practice this means trying to replace any nested for loops by calls to equivalent Numpy array methods. The goal is to avoid the CPU wasting time in the Python interpreter rather than crunching numbers to fit your statistical model.

        self.C2 = C2
        self.method = method

        else:
            return 'Untrained Auto Logistic Regression Object'
        
    def fit(self,X,y):
        num_samples, num_features = X.shape
        self.unique_ = np.unique(y) # get each unique class value
        self.classifiers_ = [] # will fill this array with binary classifiers

        if len(self.unique_)>2:
            print('multiple class, unique of target:',self.unique_)
            num_unique_classes = len(self.unique_)
            for i,yval in enumerate(self.unique_): # for each unique value
                y_binary = y==yval # create a binary problem
                # train the binary classifier for this class
                blr = L2Regularization(self.eta, self.iters, self.C2, self.method)

4.2 Implement an optimization technique for logistic regression using mean square error as your objective function¶

The derivation process

The fomula comes from class github https://github.com/eclarson/MachineLearningNotebooks/blob/master/05.%20Logistic%20Regression.ipynb

mean square error objective function:

$$ l(\mathbf{w}) = \frac{1}{2}\sum_i \left( y^{(i)} -g( \mathbf{w}^T \mathbf{x}^{(i)}) \right)^2 $$

We can simplify this calculation a bit using matrix algebra to solve for the entire vector $w$:

$$ \mathbf{w} \leftarrow \mathbf{w} + \eta \underbrace{ \sum_{i=1}^M (y^{(i)}-g(\mathbf{w}^T\mathbf{x}^{(i)})) \cdot g(\mathbf{w}^T\mathbf{x}^{(i)})(1-g(\mathbf{w}^T\mathbf{x}^{(i)})) \mathbf{x}^{(i)} }_{\text{gradient}} $$
$$ \frac{\partial}{\partial \theta} g( \theta ) = g( \theta )(1-g( \theta )) $$$$ \frac{\partial}{\partial w} g( \mathbf{w}^T \mathbf{x}^{(i)} ) = g(\mathbf{w}^T\mathbf{x}^{(i)})(1-g(\mathbf{w}^T\mathbf{x}^{(i)})) \mathbf{x}^{(i)} $$$$ \frac{\partial}{\partial w} \mathbf{w}^T \mathbf{x}^{(i)} = \mathbf{x}^{(i)} $$

mean square error objective function: $N$ is the number of instances.
for our train data, N is 1600 $$ \ell (\mathbf{w}) = \frac{1}{2N}\sum_{i=1}^N \left( y^{(i)} -g( \mathbf{w}^T \mathbf{x}^{(i)}) \right)^2 $$

class MSEBinaryLogisticRegression:
    def __init__(self, eta, iterations=100, C=0.001):
        self.eta = eta
        self.iters = iterations
        self.C = C
        
        
    def __str__(self):
        if(hasattr(self,'w_')):
            return 'MSE Binary Logistic Regression Object with coefficients:\n'+ str(self.w_) 
        else:
            return 'Untrained Binary Logistic Regression Object'
        
    # convenience, private:
    @staticmethod
    def _add_bias(X):
        return np.hstack((np.ones((X.shape[0],1)),X)) # add bias term
    
    @staticmethod
    def _sigmoid(theta):
        # increase stability, redefine sigmoid operation
        return expit(theta) #1/(1+np.exp(-theta))
    
    # vectorized gradient calculation with regularization using L2 Norm
    def _get_gradient(self,X,y):
        g = self.predict_proba(X,add_bias=False).ravel() # get sigmoid value for all classes
        hessian = X.T @ np.diag(g*(1-g)) @ X - 2 * self.C # calculate the hessian

        ydiff = y-g # get y difference
        gradient = (np.mean(X * (ydiff * g * (1-g))[:,np.newaxis], axis=0)) * 2 # make ydiff a column vector and multiply through
        gradient = gradient.reshape(self.w_.shape)
        gradient[1:] += -2 * self.w_[1:] * self.C
        
        return pinv(hessian) @ gradient
    
    # public:
    def predict_proba(self,X,add_bias=True):
        # add bias term if requested
        Xb = self._add_bias(X) if add_bias else X
        return self._sigmoid(Xb @ self.w_) # return the probability y=1
    
    def predict(self,X):
        return (self.predict_proba(X)>0.5) #return the actual prediction
    
    
    def fit(self, X, y):
        Xb = self._add_bias(X) # add bias term
        num_samples, num_features = Xb.shape
        
        self.w_ = np.zeros((num_features,1)) # init weight vector to zeros

            # for as many as the max iterations
        for _ in range(self.iters):
            gradient = self._get_gradient(Xb,y)
            self.w_ += gradient * self.eta # multiply by learning rate

Solution should be able to solve the binary logistic regression problem in one update step.¶

To solve the binary logistic regression problem in one gradient update step, we can use Newton's method $$ w \leftarrow w - \underbrace{[\frac{\partial^2}{\partial w}\ell(w)]^{-1}}_{\text{inverse 2nd deriv}}\underbrace{\frac{\partial}{\partial w}\ell(w)}_{\text{derivative}} $$
